------------

{
 stataments;
}

{
 stataments;
}

{
 stataments;
}

}


switch case
--------------
switch(expression)
{
case <condition1>: stataments; break;
case <condition2>: stataments; break;
case <condition3>: stataments; break;
default:stataments;
}
